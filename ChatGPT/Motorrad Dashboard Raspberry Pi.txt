b. Beispielcode (Play/Pause, Next, Previous):
import dbus
import time

def send_avrcp_command(command):
    bus = dbus.SessionBus()
    try:
        player = bus.get_object("org.bluez", "/org/bluez/hci0/dev_XX_XX_XX_XX_XX_XX/player0")
        iface = dbus.Interface(player, dbus_interface='org.bluez.MediaPlayer1')

        if command == "play":
            iface.Play()
        elif command == "pause":
            iface.Pause()
        elif command == "next":
            iface.Next()
        elif command == "previous":
            iface.Previous()
        else:
            print("Unbekannter Befehl")
    except dbus.exceptions.DBusException as e:
        print("Fehler:", e)

# Beispielnutzung:
send_avrcp_command("play")
time.sleep(2)
send_avrcp_command("pause")
Wichtig: Ersetze dev_XX_XX_... mit der MAC-Adresse deines Handys im Format dev_XX_XX_XX_XX_XX_XX

Tipp: Den exakten Pfad siehst du mit:

busctl tree org.bluez
